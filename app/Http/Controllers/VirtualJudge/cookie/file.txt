#include<iostream>
typedef long long LL;
using namespace std;
 
int main()
{
    LL a,b,c,d;
    LL n;
    cin>>a>>b;
    cin>>c>>d;
    cin>>n;
    LL cross=0;
    LL x,y,z;
    for(LL i=0;i<n;i++)
    {
        cin>>x>>y>>z;
        if((a*x+b*y+z)>0&&(c*x+d*y+z)<0)cross++;
        if((a*x+b*y+z)<0&&(c*x+d*y+z)>0)cross++;
    }
    cout<<cross<<endl;
    return 0;
}
